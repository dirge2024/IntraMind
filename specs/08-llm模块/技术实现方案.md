## 3. PromptBuilder

```java
@Component
public class PromptBuilder {

    public String build(String query, List<RetrievalResult> chunks, List<Message> history) {
        StringBuilder sb = new StringBuilder();
        sb.append("你是 LocalRAG 知识库助手。\n");
        sb.append("只根据【参考资料】回答，不要编造。\n\n");

        if (!chunks.isEmpty()) {
            sb.append("【参考资料】\n");
            for (var chunk : chunks) {
                sb.append(chunk.getText()).append("\n\n");
            }
        }

        if (!history.isEmpty()) {
            sb.append("【历史对话】\n");
            for (var msg : history) {
                sb.append(msg.getRole()).append("：").append(msg.getContent()).append("\n");
            }
            sb.append("\n");
        }

        sb.append("【用户问题】\n").append(query);
        return sb.toString();
    }
}
```

## 4. ChatHistoryManager

```java
@Component
public class ChatHistoryManager {
    private static final int MAX = 15;
    private final Map<String, Deque<Message>> sessions = new ConcurrentHashMap<>();

    public void append(String sessionId, String role, String content) {
        Deque<Message> history = sessions.computeIfAbsent(sessionId,
                k -> new ConcurrentLinkedDeque<>());
        history.addLast(new Message(role, content));
        while (history.size() > MAX) {
            history.pollFirst();
        }
    }

    public List<Message> getRecent(String sessionId) {
        Deque<Message> history = sessions.get(sessionId);
        return history == null ? List.of() : new ArrayList<>(history);
    }

    public record Message(String role, String content) {}
}
```

## 5. DeepSeekLlmService（SSE 流式）

```java
@Component
public class DeepSeekLlmService implements LlmService {

    private final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(10, TimeUnit.SECONDS)
            .readTimeout(60, TimeUnit.SECONDS).build();

    public SseEmitter chat(String query, String sessionId) {
        SseEmitter emitter = new SseEmitter(60000L);

        // 1. 检索
        List<RetrievalResult> chunks = retrievalService.search(query, 5);

        // 2. 历史
        List<ChatHistoryManager.Message> history = chatHistoryManager.getRecent(sessionId);

        // 3. Prompt
        String prompt = promptBuilder.build(query, chunks, history);

        // 4. 调 DeepSeek API
        String body = objectMapper.writeValueAsString(Map.of(
            "model", "deepseek-chat",
            "messages", List.of(Map.of("role", "user", "content", prompt)),
            "stream", true
        ));

        Request request = new Request.Builder()
            .url("https://api.deepseek.com/chat/completions")
            .header("Authorization", "Bearer " + config.getDeepseekApiKey())
            .post(RequestBody.create(body, JSON)).build();

        // 5. SSE 流式转发
        executor.submit(() -> {
            try (Response response = client.newCall(request).execute()) {
                BufferedReader reader = new BufferedReader(
                    new InputStreamReader(response.body().byteStream()));
                String line;
                StringBuilder fullAnswer = new StringBuilder();
                while ((line = reader.readLine()) != null) {
                    if (line.startsWith("data: ")) {
                        String json = line.substring(6);
                        if ("[DONE]".equals(json)) break;
                        String content = parseDelta(json);
                        if (content != null) {
                            fullAnswer.append(content);
                            emitter.send(SseEmitter.event().data(
                                Map.of("content", content, "done", false)));
                        }
                    }
                }
                chatHistoryManager.append(sessionId, "user", query);
                chatHistoryManager.append(sessionId, "assistant", fullAnswer.toString());
                emitter.send(SseEmitter.event().data(Map.of("content", "", "done", true)));
                emitter.complete();
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });

        return emitter;
    }
}
```
